Un query representativo

Análisis de retención por cohorte usando window functions en SQL.

-- Retención mensual por cohorte
WITH cohortes AS (
  SELECT
    user_id,
    DATE_TRUNC(created_at, MONTH) AS mes_cohorte
  FROM usuarios
),
actividad AS (
  SELECT
    user_id,
    DATE_TRUNC(evento_at, MONTH) AS mes_activo
  FROM eventos
)
SELECT
  c.mes_cohorte,
  DATE_DIFF(a.mes_activo, c.mes_cohorte, MONTH) AS periodo,
  COUNT(DISTINCT a.user_id) /
    COUNT(DISTINCT c.user_id) AS tasa_retencion
FROM cohortes c
LEFT JOIN actividad a USING(user_id)
GROUP BY 1, 2
ORDER BY 1, 2;
